
game_result = 'X' || 'O' || '0'

if (player turn)
    user_turn()
    game_result = is_game_over(turn);
    turn = game_result != 0 ? NOONE : COMPUTER_TURN
if (computer turn)
    computer_turn();
    game_result = game_result(turn)
    turn = game_result != 0 ? NOONE : PLAYER_TURN

if (is_board_full() && game_result == '0')
    print "Game is draw"
    reset_game(board, cells_state, turn)
else if (game_result == 'X')
    print "X win"
    reset_game(board, cells_state, turn)
else if (game_result == 'O')
    print "O win"
    reset_game(board, cells_state, turn)
